# DIM
dim_arr_default|DIM B[3]|B[0]|prog

dim_arr_2d|DIM POS[3,2]:POS[2,1]=9|POS[2,1]|prog
dim_scalar_init|DIM X=56|X|prog
dim_scalar_default|DIM N|N|prog

# INC
inc_default|X=3:INC X|X|prog
inc_explicit|X=3:INC X,2|X|prog

# DEC

# SWAP
swap_int|A=1:B=2:SWAP A,B|STR$(A)+STR$(B)|progstr
swap_str|A$="X":B$="Y":SWAP A$,B$|A$+B$|progstr
swap_arr|DIM A[3]:A[0]=10:A[2]=99:SWAP A[0],A[2]|STR$(A[0])+STR$(A[2])|progstr

swap_cross_numeric|A=1:B#=2.5:SWAP A,B#|STR$(A)+","+STR$(B#)|progstr
# error cases
